        const mediaChannels = "{{ media_channels }}";

        const postData = {
            media_channels: mediaChannels,
            base_variables: baseVariables,
            target_variable: targetVariable
        };

        fetch('/run_model', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify(postData),
        })
        .then(response => response.json())
        .then(data => {
            // Hide the spinner once data is received
            document.getElementById('loading-spinner-tab1').style.display = 'none';

            // Display the new model results
            displayModelResults(data);
        })
        .catch((error) => {
            console.error('Error:', error);
            // Hide the spinner in case of error
            document.getElementById('loading-spinner-tab1').style.display = 'none';
        });
    });
});

        document.querySelectorAll('input[type="range"]').forEach(slider => {
            slider.addEventListener('input', function() {
                const value = this.value;
                const max = this.max;
                const percentage = (value / max) * 100;
                const color = `linear-gradient(to right, #3B82F6 ${percentage}%, #BFDBFE ${percentage}%)`;
                this.style.background = color;
            });

            // Initialize the color based on current value
            const value = slider.value;
            const max = slider.max;
            const percentage = (value / max) * 100;
            const color = `linear-gradient(to right, #3B82F6 ${percentage}%, #BFDBFE ${percentage}%)`;
            slider.style.background = color;
        });
    });

<script>
    const sidenav = document.getElementById("mySidenav");
    const overlay = document.getElementById("overlay");

    function toggleNav() {
        if (sidenav.style.left === "0px") {
            closeNav();
        } else {
            openNav();
        }
    }

    function openNav() {
        sidenav.style.left = "0px";
        overlay.style.display = "block";
    }

    function closeNav() {
        sidenav.style.left = "-250px";
        overlay.style.display = "none";
    }

    overlay.addEventListener("click", closeNav);

    document.addEventListener("keydown", function(event) {
        if (event.key === "Escape") {
            closeNav();
        }
    });
</script>
