from bs4 import BeautifulSoup

# Load the HTML file
with open('example.html', 'r', encoding='utf-8') as file:
    soup = BeautifulSoup(file, 'html.parser')

# Find the elements with a specific class name
class_name = 'navbar-brand'

# Modify the style for all elements with the given class name
for element in soup.find_all(class_=class_name):
    # Add or modify the style
    element['style'] = 'display: none; color: red;'  # Update styles as needed
    # Optionally, you can also modify the content
    element.string = "New Navbar Brand Text"  # Update the text content

# Save the modified HTML back to a file
with open('modified_example.html', 'w', encoding='utf-8') as file:
    file.write(str(soup))

print("HTML file updated successfully.")
